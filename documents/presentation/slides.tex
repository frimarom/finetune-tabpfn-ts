\begin{frame}
\frametitle{PFN - Prior-data Fitted Networks}
    - Transformer based architecture

    - Trains on artificial tasks generated by a prior

    - Approximates posterior predictive distribution $p(y | x, D) = \int_{t} p(y|x, t)p(t, D)$

    - Prior is a generative process used to produce large amounts of training data that imitates real-world data
\end{frame}

    \item Automatic Season Features: $\Phi_{auto}(t) = (cos(2\pi f_1 t), sin(2 \pi f_1 t), …, cos(2\pi f_k t), sin(2 \pi f_k t)) \in \mathbb{R}^{2k}$

    \item Time-varying Covariates (when available): $z_{1:C+H} = [z_1, …, z_{C+H}]$ where $z_t \in \mathbb{R}^{D_{z}}$ represents $D_z$ external variables observed at time \textbf{t}
\end{itemize}

\[
X = \{x_1, x_2, \dots, x_n\} \in \mathbb{R}^{d \times n}

\begin{frame}{ForecastPFN Prior}

\small

\begin{columns}[T]

\column{0.5\textwidth}

\textbf{Global Trend}

\vspace{0.3cm}

\[
\text{trend}(t)
=
(1+m_{lin}t+c_{lin})
(m_{exp}c_{exp}^t)
\]

\vspace{0.4cm}

\[
m_{lin},m_{exp}
\sim
\mathcal{N}(\mu_m,\sigma_m)
\]

\[
c_{lin}
\sim
\mathcal{N}(0,\sigma_{lin})
\]

\[
c_{exp}
\sim
\mathcal{N}(0,\sigma_{exp})
\]

\column{0.5\textwidth}

\textbf{Noise}

\vspace{0.3cm}

\[
z_t
=
1+m_{noise}(z-\bar z)
\]

\vspace{0.4cm}

\[
z
\sim
\text{Weibull}(1,k)
\]

\[
\bar z
=
(\ln(2))^{1/k}
\]

\vspace{0.6cm}

\[
y_t
=
\text{trend}(t)\cdot z_t
\]

\end{columns}

\end{frame}

\begin{frame}{ForecastPFN Prior}

\small

\begin{columns}[T]

\column{0.58\textwidth}

\textbf{Seasonal Component}

\vspace{0.3cm}

\[
\text{seasonal}(t)
=
\prod_{\nu \in \{\text{week,month,year}\}}
\text{seasonal}_{\nu}(t)
\]

\vspace{0.4cm}

\[
\text{seasonal}_{\nu}(t)
=
1+m_\nu
\sum_f
\left[
c_{f,\nu}
\sin\left(
2\pi f \frac{t}{p_\nu}
\right)
+
d_{f,\nu}
\cos\left(
2\pi f \frac{t}{p_\nu}
\right)
\right]
\]

\column{0.42\textwidth}

\textbf{Periods}

\vspace{0.3cm}

\[
p_{week}=7
\]

\[
p_{month}=30.5
\]

\[
p_{year}=365.25
\]

\vspace{0.5cm}

\[
c_{f,\nu},d_{f,\nu}
\sim
\mathcal{N}(0,1/f)
\]

\vspace{0.6cm}

\[
y_t
=
\text{trend}(t)
\cdot
\text{seasonal}(t)
\cdot
z_t
\]

\end{columns}
