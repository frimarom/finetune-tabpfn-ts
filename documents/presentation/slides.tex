    - Approximates posterior predictive distribution $p(y | x, D) = \int_{t} p(y|x, t)p(t, D)$

\section{Conclusion}

\begin{frame}
\frametitle{Conclusion of Single Finetuning}
    \begin{itemize}
        \item finetuning on a single dataset works but is dependent on
        \begin{itemize}
            \setlength{\itemsep}{10pt}
            \item \textbf{number} of time series
            \item \textbf{context length}
            \item dataset \textbf{structure}
            \begin{itemize}
                \setlength{\itemsep}{10pt}
                \item \textbf{clear patterns} are beneficial
                \item \textbf{noisy} or \textbf{intermittent} data remain \textbf{challenging}
            \end{itemize}
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Other Two Methods}

\textbf{Multi Finetuning}
\begin{itemize}
    \item<1-> Clustering into three categories
    \begin{itemize}
        \item<2-> \textbf{Small} only led to \textbf{deterioration}
        \item<3-> \textbf{Medium} showed the \textbf{best} generalization results
        \item<4-> \textbf{Large} showed \textbf{acceptable} results
    \end{itemize}
\end{itemize}

\vspace{0.5cm}

\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Limitations}
    \begin{itemize}
        \setlength{\itemsep}{10pt}
        \item<2-> datasets only provide \textbf{limited perspective}
        \item<3-> not all \textbf{evaluation scores} could be reproduced
        \item<4-> \textbf{mismatch} between validation loss and evaluation metric
        \item<5-> computational limitations
        \item<6-> \textbf{TabPFN 2.5} was released during the thesis
    \end{itemize}
\end{frame}

\begin{frame}
\frametitle{Outlook}
    \begin{itemize}
        \setlength{\itemsep}{10pt}
        \item<2-> improving \textbf{evaluation procedure}
        \item<3-> \textbf{dataset selection} for multi-dataset finetuning
        \item<4-> systematic \textbf{hyperparameter search}
        \item<5-> effect of newer model versions (TabPFN 2.5)
        \item<6-> sensitivity of \textbf{prior-based continued pretraining}
    \end{itemize}
\end{frame}

\begin{frame}
\frametitle{Conclusion}
    \textbf{Which one of the following models performs best on various high-scale benchmarks: TabPFN-TS, finetuned TabPFN-TS, or continued pre-trained TabPFN-TS?}
    \begin{itemize}
        \setlength{\itemsep}{10pt}
        \item<2-> \textbf{no} finetuning strategy dominates all scenarios
        \item<3-> Based on \textbf{use case} for finetuning
        \item<4-> Based on \textbf{dataset availability}, \textbf{structure} and \textbf{quality} of the underlying data
        \begin{itemize}
            \item<5-> \textbf{Prior} should be used carefully
            \item<5-> Not much real world data but many different datasets $\rightarrow$ \textbf{Multi finetuning}
            \item<5-> Much real world datasets and many time series in it $\rightarrow$ \textbf{Single finetuning}
        \end{itemize}
    \end{itemize}
\end{frame}

    - Time-varying Covariates (when available): $z_{1:C+H} = [z_1, …, z_{C+H}]$ where $z_t \in \mathbb{R}^{D_{z}}$ represents $D_z$ external variables observed at time \textbf{t}
